extern void object::Solution( )
{
	while ( true )
	{
		object	spot;
		float	dir;

		spot = radar(WayPoint);
		if ( spot == null )  // plus rien ?
		{
			motor(0, 0);  // stoppe
			break;
		}

		dir = direction(spot.position);
		if ( dir < 0 )  // à droite ?
		{
			motor(1, 1+dir/90);
		}
		else  // à gauche ?
		{
			motor(1-dir/90, 1);
		}
	}
}
