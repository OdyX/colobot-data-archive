extern void object::Solution()
{
	object    chose;
	
	aim(-20);
	
	while ( true )
	{
		while ( radar(AlienAnt, 0, 360, 0, 20) == null )
		{
			chose = radar(AlienAnt);
			turn(direction(chose.position));
			motor(1,1);
			jet(0);

			if ( position.z-topo(position) < 6 )
			{
				jet(1);
			}
			
			if ( position.z-topo(position) > 9 )
			{
				jet(-1);
			}

			wait(0.2);
		}
		
		fire(1);
	}
}
