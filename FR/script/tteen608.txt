extern void object::Solution( )
{
	repeat(100)
	{
		if( detect(Scrap) )
		{
			grab();
			turn(180);
			drop();
			turn(180);
		}
		
		if( not detect(Barrier) )
		{
			move(20);
			continue;
		}
		
		turn(-90);
		if( not detect(Barrier) )
		{
			move(20);
			continue;
		}
		
		turn(180);
		if( not detect(Barrier) )
		{
			move(20);
			continue;
		}
	}
}
