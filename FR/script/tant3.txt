extern void object::Solution()
{
	object    chose;
	
	jet(0.2);
	while ( position.z < 20 )
	{
		wait(0.2);
	}
	jet(0);
	
	while ( true )
	{
		chose = radar(AlienAnt);
		turn(direction(chose.position));
		while ( radar(AlienAnt, 0, 360, 0, 40) == null )
		{
			wait(0.2);
		}
		fire(1);
	}
}
