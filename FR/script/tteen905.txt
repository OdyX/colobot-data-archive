extern void object::Solution( )
{
	int gauche = 90;
	int droite = -90;
	int couleur = Blue;
	
	repeat(6)
	{
		pendown(couleur);
		
		if( couleur == Blue )
		{
			couleur = Red;
		}
		else
		{
			couleur = Blue;
		}
		
		move(7);
		turn(droite);
		move(7);
		turn(gauche);
	}
}
